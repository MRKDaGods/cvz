% Entry commands use tabular* for precise left/right alignment (FAANG standard).
% \expentry{Company}{Role}{Location}{Dates}

% \achievementbox{content} — light callout for key career wins

\newcommand{\expentry}[4]{%
  \vspace{0pt}%
  \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
    \textbf{#1} & \textit{#4} \\%
    \textit{\small #2} & {\small #3} \\%
  \end{tabular*}\par\nopagebreak\vspace{-4pt}%
}

\newcommand{\eduentry}[4]{%
  \vspace{0pt}%
  \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
    \textbf{#1} & \textit{#4} \\%
    \textit{\small #2} & {\small #3} \\%
  \end{tabular*}\par\nopagebreak\vspace{-4pt}%
}

\newcommand{\projentry}[3]{%
  \vspace{0pt}%
  \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
    \textbf{#1}\enspace---\enspace\textit{#2} & #3 \\%
  \end{tabular*}\par\nopagebreak\vspace{-4pt}%
}

% Key achievements callout box — ATS-parseable (simple \fcolorbox)
\newcommand{\achievementbox}[1]{%
  \vspace{3pt}%
  \noindent\fcolorbox{rulecolor}{achievebg}{%
    \begin{minipage}{\dimexpr\linewidth-2\fboxsep-2\fboxrule}
      #1
    \end{minipage}%
  }%
  \vspace{3pt}%
}
