% MODERN CV TEMPLATE — Clean Accent-Colored Layout

% Single-column layout with a distinctive navy accent color on headings and
% a clean geometric sans-serif font. Strikes the balance between visual
% personality and ATS-friendliness.
% Font: Cabin (geometric humanist sans-serif — distinctive yet professional)
%
% Compile with: latexmk -pdf modern.tex
%
% ╔═══════════════════════════════════════════════════════════════════════════╗
% ║  GUARDRAILS — READ BEFORE MODIFYING                                     ║
% ║  • Do NOT add \definecolor, \colorlet, or any color commands not        ║
% ║    already defined below.                                               ║
% ║  • Do NOT add packages not listed below (no tikz, tcolorbox, etc.).     ║
% ║  • Do NOT \renewcommand or redefine any command defined below.          ║
% ║  • Do NOT add sidebar backgrounds, column coloring, or shading.         ║
% ║  • Do NOT add \titleformat, \titlespacing, \setlist, or \pagestyle      ║
% ║    overrides — the template handles all styling.                        ║
% ║  • The ONLY acceptable modifications are filling {{PLACEHOLDER}} values ║
% ║    and removing entire sections that have no data.                       ║
% ╚═══════════════════════════════════════════════════════════════════════════╝
%
% PLACEHOLDER MARKERS:
%   {{NAME}}            — Full name
%   {{EMAIL}}           — Email address
%   {{PHONE}}           — Phone number
%   {{LINKEDIN}}        — LinkedIn username
%   {{GITHUB}}          — GitHub username
%   {{LOCATION}}        — City, State or City, Country
%   {{SUMMARY}}         — Professional summary paragraph
%   {{EXPERIENCE}}      — Full experience block
%   {{EDUCATION}}       — Full education block
%   {{SKILLS}}          — Skills content (compact categorized list)
%   {{PROJECTS}}        — Projects block
% =============================================================================

\documentclass[10pt,letterpaper]{article}

% ---------------------------------------------------------------------------
% PACKAGES — Do NOT add any packages beyond these.
% ---------------------------------------------------------------------------

\usepackage[
  letterpaper,
  top=0.4in,
  bottom=0.4in,
  left=0.5in,
  right=0.5in
]{geometry}

\usepackage{titlesec}
\usepackage{enumitem}
\usepackage{fontawesome5}
\usepackage{microtype}

% Font: Cabin — geometric humanist sans-serif, visually distinct from Helvetica
\usepackage[sfdefault]{cabin}
\usepackage[T1]{fontenc}

\usepackage{xcolor}

% ---------------------------------------------------------------------------
% COLORS — Do NOT modify, redefine, or add new colors.
% These are the ONLY colors in this template.
% ---------------------------------------------------------------------------

\definecolor{accent}{HTML}{2C3E50}      % Dark navy — headings and name
\definecolor{rulecolor}{HTML}{BDC3C7}   % Section rule lines
\definecolor{linkblue}{HTML}{2C3E50}    % URL link color (matches accent)

\usepackage[
  colorlinks=true,
  urlcolor=linkblue,
  linkcolor=linkblue,
  pdfborder={0 0 0}
]{hyperref}

% ATS: ensure generated PDF text is machine-readable
\input{glyphtounicode}
\pdfgentounicode=1

% ---------------------------------------------------------------------------
% STYLE CONFIGURATION — Do NOT modify these settings.
% ---------------------------------------------------------------------------

\pagestyle{empty}
\setlength{\parindent}{0pt}
\setlength{\parskip}{0pt}
\setlength{\tabcolsep}{0in}
\raggedbottom
\raggedright

% Section headings — accent-colored small caps with colored rule
\titleformat{\section}
  {\color{accent}\scshape\raggedright\large}
  {}
  {0em}
  {}
  [\vspace{1pt}{\color{rulecolor}\titlerule}]

\titlespacing*{\section}
  {0pt}{6pt}{5pt}

% Tight bullet lists
\setlist[itemize]{
  nosep,
  leftmargin=1.3em,
  label=\textbullet,
  topsep=0pt,
  parsep=0pt,
  itemsep=1pt,
  partopsep=0pt
}

% ---------------------------------------------------------------------------
% CUSTOM COMMANDS — Do NOT redefine or renewcommand these.
%
% Entry commands use tabular* for precise left/right alignment (FAANG standard).
% Company names rendered in accent color for visual distinction.
% \expentry{Company}{Role}{Location}{Dates}
% \eduentry{School}{Degree}{Location}{Dates}
% \projentry{Name}{Tech Stack}{Date/Link}
% ---------------------------------------------------------------------------

\newcommand{\expentry}[4]{%
  \vspace{2pt}%
  \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
    {\color{accent}\textbf{#1}} & \textit{#4} \\%
    \textit{\small #2} & {\small #3} \\%
  \end{tabular*}\par\nopagebreak\vspace{-3pt}%
}

\newcommand{\eduentry}[4]{%
  \vspace{2pt}%
  \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
    {\color{accent}\textbf{#1}} & \textit{#4} \\%
    \textit{\small #2} & {\small #3} \\%
  \end{tabular*}\par\nopagebreak\vspace{-3pt}%
}

\newcommand{\projentry}[3]{%
  \vspace{2pt}%
  \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
    {\color{accent}\textbf{#1}}\enspace---\enspace\textit{#2} & #3 \\%
  \end{tabular*}\par\nopagebreak\vspace{-3pt}%
}

% =============================================================================

% =============================================================================

\begin{document}

% ---------------------------------------------------------------------------
% HEADER — accented name with icon-based contact line
% ---------------------------------------------------------------------------
\begin{center}
  {\color{accent}\Huge\bfseries {{NAME}}}\\[4pt]
  {\color{rulecolor}\rule{0.4\textwidth}{0.4pt}}\\[4pt]
  \small
  \faIcon{phone}\enspace{}{{PHONE}} \quad
  \href{mailto:{{EMAIL}}}{\faIcon{envelope}\enspace{}{{EMAIL}}} \quad
  \href{https://linkedin.com/in/{{LINKEDIN}}}{\faIcon{linkedin}\enspace{}{{LINKEDIN}}} \quad
  \href{https://github.com/{{GITHUB}}}{\faIcon{github}\enspace{}{{GITHUB}}} \quad
  \faIcon{map-marker-alt}\enspace{}{{LOCATION}}
\end{center}

\vspace{3pt}
