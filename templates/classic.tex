% ---------------------------------------------------------------------------
% CUSTOM COMMANDS — Do NOT redefine or renewcommand these.
%
% \expentry{Company}{Role}{Location}{Dates}
%   Renders: ROLE | COMPANY (bold, line 1)
%            Location (italic, left) — Dates (bold, right-aligned, line 2)
%
% \eduentry{School}{Degree}{Location}{Dates}  — same layout
% \projentry{Name}{Tech Stack}{Date}
% \awardentry{Award Name}{Issuer}{Date}
% ---------------------------------------------------------------------------

\newcommand{\expentry}[4]{%
  \par\nobreak\vspace{4pt}%
  \textbf{#2 | #1}\\[1pt]%
  \textit{#3}\hfill\textbf{#4}%
  \par\nopagebreak\vspace{1pt}%
}

\newcommand{\eduentry}[4]{%
  \par\nobreak\vspace{4pt}%
  \textbf{#2 | #1}\\[1pt]%
  \textit{#3}\hfill\textbf{#4}%
  \par\nopagebreak\vspace{1pt}%
}

\newcommand{\projentry}[3]{%
  \par\nobreak\vspace{4pt}%
  \textbf{#1}\enspace\textit{| #2}\hfill\textbf{#3}%
  \par\nopagebreak\vspace{1pt}%
}

\newcommand{\awardentry}[3]{%
  \par\nobreak\vspace{3pt}%
  \textbf{#1}\enspace|\enspace#2\hfill\textit{#3}%
  \vspace{2pt}%
}
