% Experience: \expentry{Company}{Role}{Location}{Dates}
\newcommand{\expentry}[4]{%
  \vspace{5pt}%
  {\raggedright\textbf{#1}\hfill\textit{#4}}\par\nopagebreak
  {\raggedright\textit{#2}\hfill#3}\par\nopagebreak
  \vspace{1pt}%
}

% Education: \eduentry{School}{Degree}{Location}{Dates}
\newcommand{\eduentry}[4]{%
  \vspace{5pt}%
  {\raggedright\textbf{#1}\hfill\textit{#4}}\par\nopagebreak
  {\raggedright\textit{#2}\hfill#3}\par\nopagebreak
  \vspace{1pt}%
}

% Project: \projentry{Name}{Tech Stack}{Date/Link}
\newcommand{\projentry}[3]{%
  \vspace{5pt}%
  {\raggedright\textbf{#1}\enspace---\enspace\textit{#2}\hfill#3}\par\nopagebreak
  \vspace{1pt}%
}
